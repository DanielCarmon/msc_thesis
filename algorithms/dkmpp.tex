\begin{algorithm}[H]
\caption{Deep K-Means++}

	\State \textbf{Input:} Embedded Data $\widetilde{X}$, Number of clusters $k$, Unroll depth $l$.

    \State $C_i \gets $ Infer-Centroids($\widetilde{X},B_{i-1}$)
    \EndFor
